# ISM2411 Interactive Course Map Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create `ism2411/pages/course_map.html` — a 5-unit interactive course map matching ISM3232's visual pattern, with full light/dark theme support and a nav link in the Start Here dropdown.

**Architecture:** Single self-contained HTML file with inline CSS (dark default + `[data-theme="light"]` overrides) and inline JS (DETAILS data, toggle, keyboard nav). Loads `../assets/css/site.css` for nav styles and unit color variables; loads `../assets/js/site.js` to inject the sticky nav and theme toggle. One additional `<a>` added to the Start Here dropdown in `site.js`.

**Tech Stack:** Vanilla HTML/CSS/JS, JetBrains Mono, CSS custom properties, CSS `color-mix()`, `grid-template-columns` for 5-unit layout.

let activeUnit = null;

function toggle(el) {
  const uid = parseInt(el.dataset.unit);
  const detail = document.getElementById('detail');
  const inner  = document.getElementById('detail-inner');

  document.querySelectorAll('.unit').forEach(u => u.classList.remove('active'));

  if (activeUnit === uid) {
    activeUnit = null;
    detail.classList.remove('open');
    detail.style.borderColor = '';
    return;
  }

  activeUnit = uid;
  el.classList.add('active');

  const d = DETAILS[uid];
  inner.innerHTML = d.cols.map(col => `
    <div class="detail-col">
      <div class="detail-label" style="color:${d.color}">${col.label}</div>
      <div class="detail-items">
        ${col.items.map(i => `<div class="detail-item">${i}</div>`).join('')}
      </div>
    </div>
  `).join('');

  detail.style.borderColor = `color-mix(in srgb, ${d.color} 40%, transparent)`;
  detail.classList.add('open');
  animateArrows(uid);
}

function animateArrows(uid) {
  for (let i = 1; i <= 4; i++) {
    document.getElementById('arr' + i).classList.toggle('active', i < uid);
  }
}

document.addEventListener('keydown', e => {

    document.getElementById('detail').classList.remove('open');
    document.querySelectorAll('.arrow-col').forEach(a => a.classList.remove('active'));

  if (['1','2','3','4','5'].includes(e.key)) {
    toggle(document.querySelectorAll('.unit')[parseInt(e.key) - 1]);
  }
});
